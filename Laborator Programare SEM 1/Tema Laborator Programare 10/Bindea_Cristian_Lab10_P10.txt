int* citire(int n)
{
	static int v[DIM];
	for (int i = 0; i < n; i++)
	{
		printf("\ta[%d] = ", i);
		scanf("%d", v+i);
	}

	return v;
}
int* completare(int* p, int n)
{
	int media = 0;
	static int v2[DIM];
	for (int i = 0; i < n; i++)
		media += p[i];
	media /= n;
	for (int i = 0; i < n; i++)
		v2[i] = p[i] - media;
	
	return v2;
}
